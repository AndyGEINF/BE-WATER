249 public static float cabal(Nodo nodo, Graph graph, Xarxa x){
250 if(graph.getNode(nodo.id()) == null){//No pertany a la xarxa
251 throw new NoSuchElementException("No pertany a la xarxa " + nodo.id());
252 }
253 if (!nodo.aixetaOberta()) { //aixeta tancada
254 return 0;
255 }
256 if(nodo instanceof Origen){ //és un node origen
257 Origen o = ((Origen)nodo);

259 if(demanda>o.cabal()){
260 return o.cabal();
261 }
262 else{
263 return demanda;
264 }

267 Stream<Edge> entrades=graphNode.enteringEdges();
268 Iterable<Edge> iteradorEntrades = entrades::iterator;
269 float cabal=0;
270 for (Edge edge : iteradorEntrades) {
271 cabal+=cabalEntrant(edge, graph, x);
272 }
273 return cabal;

563 private static Boolean RhiHaCami(Node n1,Node n2,List<String> visitats){
564 if(n1.hasEdgeToward(n2) && fluxQuePotPortar(n1.getEdgeToward(n2))>0){
565 visitats.add(n2.getId());
566 return true;
567 }
568 else{
569 Boolean trobat=false;
570 Stream<Edge> arestesSortida = n1.leavingEdges();
571 Iterator<Edge> itS=arestesSortida.iterator();
572 while(itS.hasNext() && !trobat){//sortides mentres no s'hagi arribat a terminal
573 Edge e = itS.next();
574 Float fluxPossible=fluxQuePotPortar(e);
575 if(fluxPossible>0 && !visitats.contains(e.getTargetNode().getId())){//si no es pot portar flux no hi anem
576 visitats.add(e.getTargetNode().getId());
577 trobat=RhiHaCami(e.getTargetNode(), n2, visitats);
578 if(!trobat){
579 visitats.remove(e.getTargetNode().getId());
580 }
581 }

694 private static void posarArestes(Xarxa x, Graph graphMaxFlow, Iterator<Canonada> it, List<String> origensVisitats, List<String> terminalsVisitats){
695 //poso els nodes si fa falta, després poso l'aresta amb la capacitat
696 // corresponent, tenint en compte que si el superorigen o superterminal
697 // ja estaven connectats a un node connexió la capacitat es suma.
698 while(it.hasNext()){
699 Canonada c=it.next();
700 Nodo n1=c.node1();
701 Nodo n2=c.node2();
702 //poso els nodes si son Connexions
703 if(n1 instanceof Connexio && graphMaxFlow.getNode(n1.id())==null){//Si és connexio i no pertany a la xarxa
704 graphMaxFlow.addNode(n1.id());
705 posarArestes(x, graphMaxFlow, x.entrades(n1), origensVisitats, terminalsVisitats);
706 posarArestes(x, graphMaxFlow, x.sortides(n1), origensVisitats, terminalsVisitats);
707 }
708 if(n2 instanceof Connexio && graphMaxFlow.getNode(n2.id())==null){//Si és connexio i no pertany a la xarxa
709 graphMaxFlow.addNode(n2.id());
710 posarArestes(x, graphMaxFlow, x.entrades(n2), origensVisitats, terminalsVisitats);
711 posarArestes(x, graphMaxFlow, x.sortides(n2), origensVisitats, terminalsVisitats);
712 }
713 if(!(n1 instanceof Origen) && !(n2 instanceof Terminal)){//tots són conexio
714 if(graphMaxFlow.getEdge(c.id())==null){//aquella aresta no existeix encara
715 Edge e=graphMaxFlow.addEdge(c.id(), n1.id(), n2.id(), true);
716 e.setAttribute(atributCapacitat, c.capacitat());
717 Edge e2=graphMaxFlow.addEdge(c.id()+"AUX", n2.id(), n1.id(), true);
718 e2.setAttribute(atributFlux,0f);//inicialitzem a 0 el flux
719 }
720 }
721 if(n1 instanceof Origen && !origensVisitats.contains(n1.id())){
722 //posar totes les sortides
723 origensVisitats.add(n1.id());
724 posarSortidesMaxFlow(x, graphMaxFlow, n1, origensVisitats, terminalsVisitats);
725 }
726 if(n2 instanceof Terminal && !terminalsVisitats.contains(n2.id())){
727 //posar totes les entrades
728 terminalsVisitats.add(n2.id());
729 posarEntradesMaxFlow(x, graphMaxFlow, n2, origensVisitats, terminalsVisitats);
730 }
731 }
