339 public float cabal(Nodo nodo){
340 if(graph.getNode(nodo.id()) == null){//No pertany a la xarxa
341 throw new NoSuchElementException("No pertany a la xarxa " + nodo.id());
342 }
343 if (!nodo.aixetaOberta()) { //aixeta tancada
344 return 0;
345 }
346 if(nodo instanceof Origen){ //es un node origen
347 // si la demanda es més gran que el cabal retorna el cabal, altrament retorna la demanda
348 Origen o = (Origen)nodo;
349 float demanda = demanda(nodo);
350 if(demanda>o.cabal()){
351 return o.cabal();
352 }
353 return demanda;
354 }
355 //no es un origen
356 Iterator<Canonada> itEntrades=entrades(nodo);
357 float cabal=0;
358 while (itEntrades.hasNext()) {
359 cabal+=cabalEntrant(itEntrades.next());
360 }
361 return cabal;

369 private float cabalEntrant(Canonada edge){

416 if(demanda>=capacitatsCanonades){
